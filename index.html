Apps Script Code (Copy This):

function doPost(e) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    const data = JSON.parse(e.postData.contents);
    
    // Find or create row for this user/week
    const allData = sheet.getDataRange().getValues();
    let rowIndex = -1;
    
    for (let i = 1; i < allData.length; i++) {
      if (allData[i][0] == data.week && allData[i][1] == data.userName) {
        rowIndex = i + 1;
        break;
      }
    }
    
    if (rowIndex === -1) {
      sheet.appendRow([
        data.week,
        data.userName,
        data.score,
        JSON.stringify(data.goals),
        new Date()
      ]);
    } else {
      sheet.getRange(rowIndex, 3).setValue(data.score);
      sheet.getRange(rowIndex, 4).setValue(JSON.stringify(data.goals));
      sheet.getRange(rowIndex, 5).setValue(new Date());
    }
    
    return ContentService.createTextOutput(JSON.stringify({success: true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({success: false, error: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const data = sheet.getDataRange().getValues();
  
  const results = [];
  for (let i = 1; i < data.length; i++) {
    if (data[i][0]) {
      results.push({
        week: data[i][0],
        userName: data[i][1],
        score: data[i][2],
        goals: JSON.parse(data[i][3] || '[]'),
        timestamp: data[i][4]
      });
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify(results))
    .setMimeType(ContentService.MimeType.JSON);
}
